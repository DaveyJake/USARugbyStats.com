<VirtualHost *:80>
  ServerName localhost
  DocumentRoot "%TRAVIS_BUILD_DIR%/public"

  <Directory "%TRAVIS_BUILD_DIR%/public">
    Options FollowSymLinks MultiViews ExecCGI
    AllowOverride All
    
    Order allow,deny
    Allow from all
    Require all granted
    Satisfy Any
  </Directory>

  # Wire up Apache to use Travis CI's HHVM  
  <IfModule mod_fastcgi.c>
    Alias /hhvm.fastcgi /usr/lib/cgi-bin/hhvm.fastcgi
    FastCGIExternalServer /usr/lib/cgi-bin/hhvm.fastcgi -socket /var/run/hhvm/socket -pass-header Authorization -idle-timeout 300
    <Directory "/usr/lib/cgi-bin">
        <Files "hhvm.fastcgi">
            Order deny,allow
        </Files>
    </Directory>

    AddHandler hhvm-hack-extension .hh
    AddHandler hhvm-php-extension .php

    Action hhvm-hack-extension /hhvm.fastcgi virtual
    Action hhvm-php-extension /hhvm.fastcgi virtual
  </IfModule>

  ## Logging
  ErrorLog "/tmp/apache2_default_error.log"
  ServerSignature Off
  CustomLog "/tmp/apache2_default_access.log" combined
</VirtualHost>
